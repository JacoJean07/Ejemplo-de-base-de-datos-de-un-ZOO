

CREATE DEFINER=`root`@`localhost` PROCEDURE `SP_S_ZoologicosParametro`(IN `pcriterio` VARCHAR(20), IN `pbusqueda` VARCHAR(20))
BEGIN
	IF pcriterio = "IDVENDEDOR" THEN
		SELECT cc.IDZOOLOGICO, cc.ZOONOMBRE, cc.ZOOCIUDAD, cc.ZOOPAIS, cc.ZOOTAMANIO, cc.ZOOPRESUPUESTOANUAL
		FROM zoologicos AS cc
		WHERE cc.IDZOOLOGICO LIKE CONCAT("%",pbusqueda,"%");
	ELSEIF pcriterio = "ZOONOMBRE" THEN
		SELECT cc.IDZOOLOGICO, cc.ZOONOMBRE, cc.ZOOCIUDAD, cc.ZOOPAIS, cc.ZOOTAMANIO, cc.ZOOPRESUPUESTOANUAL
		FROM zoologicos AS cc
		WHERE cc.ZOONOMBRE LIKE CONCAT("%",pbusqueda,"%");
   ELSEIF pcriterio = "ZOOCIUDAD" THEN
		SELECT cc.IDZOOLOGICO, cc.ZOONOMBRE, cc.ZOOCIUDAD, cc.ZOOPAIS, cc.ZOOTAMANIO, cc.ZOOPRESUPUESTOANUAL
		FROM zoologicos AS cc
		WHERE cc.ZOOCIUDAD LIKE CONCAT("%",pbusqueda,"%");
	ELSEIF pcriterio = "ZOOPAIS" THEN
		SELECT cc.IDZOOLOGICO, cc.ZOONOMBRE, cc.ZOOCIUDAD, cc.ZOOPAIS, cc.ZOOTAMANIO, cc.ZOOPRESUPUESTOANUAL
		FROM zoologicos AS cc
		WHERE cc.ZOOPAIS LIKE CONCAT("%",pbusqueda,"%");
	ELSE
		SELECT cc.IDZOOLOGICO, cc.ZOONOMBRE, cc.ZOOCIUDAD, cc.ZOOPAIS, cc.ZOOTAMANIO, cc.ZOOPRESUPUESTOANUAL
		FROM zoologicos AS cc
		WHERE cc.IDZOOLOGICO =10;
	END IF;
	END
